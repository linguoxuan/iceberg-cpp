34struct RetryConfig {
36 int32_t num_retries = 4;
38 int32_t min_wait_ms = 100;

72 template <typename F, typename T = typename std::invoke_result_t<F>::value_type>
73 Result<T> Run(F&& task, int32_t* attempt_counter = nullptr) {
74 auto start_time = std::chrono::steady_clock::now();
75 int32_t attempt = 0;
76 int32_t max_attempts = config_.num_retries + 1;
77
78 while (true) {
79 ++attempt;
80 if (attempt_counter != nullptr) {
81 *attempt_counter = attempt;
82 }
83
84 auto result = task();
85 if (result.has_value()) {
86 return result;
87 }
88
89 const auto& error = result.error();
90
91 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
92 std::chrono::steady_clock::now() - start_time)
93 .count();
94
95 // total_timeout_ms <= 0 means no total timeout limit
96 bool timed_out = config_.total_timeout_ms > 0 && elapsed > config_.total_timeout_ms;
97 if (attempt >= max_attempts || timed_out) {
98 return result;
99 }
100
101 if (!ShouldRetry(error.kind)) {
102 return result;
103 }
104
105 int32_t delay_ms = CalculateDelay(attempt);
106 Sleep(delay_ms);
107 }
108 }
109
110 private:
112 bool ShouldRetry(ErrorKind kind) const {
113 if (!only_retry_on_.empty()) {
114 return std::ranges::any_of(only_retry_on_,
115 [kind](ErrorKind k) { return kind == k; });
116 }
117
118 if (!stop_retry_on_.empty()) {
119 return !std::ranges::any_of(stop_retry_on_,
120 [kind](ErrorKind k) { return kind == k; });
121 }
122
123 return true;
124 }
125
127 int32_t CalculateDelay(int32_t attempt) const {
128 // Calculate base delay with exponential backoff
129 double base_delay = config_.min_wait_ms * std::pow(config_.scale_factor, attempt - 1);
130 int32_t delay_ms = static_cast<int32_t>(
131 std::min(base_delay, static_cast<double>(config_.max_wait_ms)));
132
133 static thread_local std::mt19937 gen(std::random_device{}());
134 int32_t jitter_range = std::max(1, delay_ms / 10);
135 std::uniform_int_distribution<> dis(0, jitter_range - 1);
136 delay_ms += dis(gen);
137 return std::max(1, delay_ms);
138 }
139
141 void Sleep(int32_t ms) const {
142 std::this_thread::sleep_for(std::chrono::milliseconds(ms));
143 }
144
145 RetryConfig config_;
146 std::vector<ErrorKind> only_retry_on_;
147 std::vector<ErrorKind> stop_retry_on_;
148};
