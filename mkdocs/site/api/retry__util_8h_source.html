88 template <typename F, typename T = typename std::invoke_result_t<F>::value_type>
89 Result<T> Run(F&& task, int32_t* attempt_counter = nullptr) {
90 if (config_.num_retries < 0) {
91 return InvalidArgument("num_retries must be non-negative, got {}",
92 config_.num_retries);
93 }
94
95 auto start_time = std::chrono::steady_clock::now();
96 int32_t attempt = 0;
97 int32_t max_attempts = config_.num_retries + 1;
98
99 while (true) {
100 ++attempt;
101 if (attempt_counter != nullptr) {
102 *attempt_counter = attempt;
103 }
104
105 auto result = task();
106 if (result.has_value()) {
107 return result;
108 }
109
110 const auto& error = result.error();
111
112 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
113 std::chrono::steady_clock::now() - start_time)
114 .count();
115
116 // total_timeout_ms <= 0 means no total timeout limit
117 bool timed_out = config_.total_timeout_ms > 0 && elapsed > config_.total_timeout_ms;
118 if (attempt >= max_attempts || timed_out) {
119 return result;
120 }
121
122 if (!ShouldRetry(error.kind)) {
123 return result;
124 }
125
126 int32_t delay_ms = CalculateDelay(attempt);
127 std::this_thread::sleep_for(std::chrono::milliseconds(delay_ms));
128 }
129 }
130
131 private:
133 bool ShouldRetry(ErrorKind kind) const {
134 if (!only_retry_on_.empty()) {
135 return std::ranges::any_of(only_retry_on_,
136 [kind](ErrorKind k) { return kind == k; });
137 }
138
139 if (!stop_retry_on_.empty()) {
140 return !std::ranges::any_of(stop_retry_on_,
141 [kind](ErrorKind k) { return kind == k; });
142 }
143
144 return true;
145 }
146
148 int32_t CalculateDelay(int32_t attempt) const {
149 // Calculate base delay with exponential backoff
150 double base_delay = config_.min_wait_ms * std::pow(config_.scale_factor, attempt - 1);
151 int32_t delay_ms = static_cast<int32_t>(
152 std::min(base_delay, static_cast<double>(config_.max_wait_ms)));
153
154 static thread_local std::mt19937 gen(std::random_device{}());
155 int32_t jitter_range = std::max(1, delay_ms / 10);
156 std::uniform_int_distribution<> dis(0, jitter_range - 1);
157 delay_ms += dis(gen);
158 return std::max(1, delay_ms);
159 }
160
161 RetryConfig config_;
162 std::vector<ErrorKind> only_retry_on_;
163 std::vector<ErrorKind> stop_retry_on_;
164};
